# 如何移除Kotlin代码中所有的!!(非空断言)


# 从今往后一万年 你们都会记住我的名字 “空安全” KotlinNullPointerException这个异常我今天就从你的字典里删除了

# 1. val替代var

Kotlin让你在语言的层面去考虑不变性，这点看起来很不错。 val 是只读，var 是可变。建议你尽可能多的使用只读属性


# 2. lateinit

有时候你不能使用不变属性。例如，在Android中onCreate方法被调用时，一些属性被初始化。在这些场景中，Kotlin有个语言特性叫做 lateinit

# 使用!!的代码:
private var mAdapter: RecyclerAdapter<Transaction>? = null

override fun onCreate(savedInstanceState: Bundle?) {
   super.onCreate(savedInstanceState)
   mAdapter = RecyclerAdapter(R.layout.item_transaction)
}

fun updateTransactions() {
   mAdapter!!.notifyDataSetChanged()
}
# lateinit写法:
private lateinit var mAdapter: RecyclerAdapter<Transaction>

override fun onCreate(savedInstanceState: Bundle?) {
   super.onCreate(savedInstanceState)
   mAdapter = RecyclerAdapter(R.layout.item_transaction)
}

fun updateTransactions() {
   mAdapter.notifyDataSetChanged()
}

# 3. let函数

  1. let: it表示引用对象,可调用其成员(属性以及方法),it不可省略.
  2. 返回值: 返回值为函数块的最后一行,为空就返回一个Unit类型的默认值.
private var mPhotoUrl: String? = null

    fun uploadClicked() {
        mPhotoUrl?.let { uploadPhoto(it) }
}

# 4. Elvis操作符 ? :

Elvis操作符作用在于如果你有空类型情况出现，且有返回值的功能。

# 比如以下代码:
fun getUserName(): String {
   if (mUserName != null) {
       return mUserName!!
   } else {
       return "Anonymous"
   }
}
# 可以被替代为如下代码:
fun getUserName(): String {
   return mUserName ?: "Anonymous"
}

# 5. 创建全局函数

let 函数是一个很好的简单检查空类型替代方式。

# 例如更多复杂的cases :
if (mUserName != null && mPhotoUrl != null) {
   uploadPhoto(mUserName!!, mPhotoUrl!!)
}
# 可以使用两个let函数嵌套调用，但是那样可读性很差。在Kotlin中你可以创建全局访问函:
ifNotNull(mUserName, mPhotoUrl) {
   userName, photoUrl ->
   uploadPhoto(userName, photoUrl)
}
# 函数定义:
fun <T1, T2> ifNotNull(value1: T1?, value2: T2?, bothNotNull: (T1, T2) -> (Unit)) {
   if (value1 != null && value2 != null) {
       bothNotNull(value1, value2)
   }
}

# 6. 自己声明崩溃

尽管你知道类型是可空的，但是有些情况下你知道一些属性是不可能为空的。一旦为空了，你应该很容易知道这是一个bug.然而抛弃使用 !! 非空断言。

# 类似如下代码:
uploadPhoto(intent.getStringExtra("PHOTO_URL")!!)
# 以上代码可以替代为:
uploadPhoto(requireNotNull(intent.getStringExtra("PHOTO_URL"), { "Activity parameter 'PHOTO_URL' is missing" }))
