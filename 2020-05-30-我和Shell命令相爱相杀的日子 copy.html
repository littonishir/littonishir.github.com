# 文件操作

sed命令常用的两个选项：
-i : 直接在文件上编辑
-e :只在命令行输出，而文件不改变

# 删除指定行

# 删除第N行
sed -i "" "Nd" filename.txt
# 删除第N~M行
sed -i "" "N,Md" filename.txt
# 删除shell变量表示的行号
sed -i "" "${var1},${var2}d" filename.txt
# 删除最后一行
sed -i "" "$d" filename.txt
# sed命令删除包含特定字符行(删除包含xxx的行)
sed -i "" "/xxx/d" filename.txt

# 匹配目标字符串替换

# 2s : 表示第二行 
# home : 目标字符串
# lastModified:\"$myDate\"  : 替换的字符串
# /g : 行内全面替换
sed -i "" "2s/home/lastModified:\"$myDate\"/g" filename.txt
